File 1: lib/utils/circuit-breaker.ts
Purpose: Track repeated failures of an external API and "open" the circuit after a threshold so subsequent calls fail fast instead of waiting for timeouts. After a cooldown, the circuit "half-opens" to test recovery. This prevents cascading failures during sustained vendor outages.
Required exports:

class CircuitBreakerError extends Error (for fail-fast rejections when circuit is open)
interface CircuitBreakerOptions (failureThreshold: number, resetTimeoutMs: number, halfOpenMaxAttempts: number)
class CircuitBreaker (with methods execute<T>(fn: () => Promise<T>): Promise<T>, getState(): 'closed' | 'open' | 'half-open', getStats(): { failures: number; successes: number; lastFailureAt: Date | null })
function getCircuitBreaker(name: string, options?: CircuitBreakerOptions): CircuitBreaker (returns a named singleton; multiple call sites for the same vendor share state)
function isCircuitBreakerError(error: unknown): error is CircuitBreakerError

Required behavior:

Default options: failureThreshold: 5 consecutive failures, resetTimeoutMs: 60000 (1 minute cooldown), halfOpenMaxAttempts: 1
States: closed (normal operation), open (failing fast), half-open (testing one request to see if upstream recovered)
When closed: passes calls through. On success, resets failure counter. On failure, increments counter; if it reaches threshold, transitions to open and starts the reset timer
When open: rejects immediately with CircuitBreakerError for any call. After resetTimeoutMs elapses, transitions to half-open
When half-open: allows one call through. On success, transitions to closed. On failure, transitions back to open and resets the timer
Singletons by name: getCircuitBreaker('claude-api') always returns the same breaker, so multiple call sites coordinate failure state

The file must include JSDoc with at least three usage examples covering: a wrap around a Claude API call, a wrap around a regulatory API call with custom thresholds, and graceful fallback when the circuit is open.

typescript// app/api/payments/process/route.ts (illustrative)
import { NextRequest, NextResponse } from 'next/server'
import { withTimeout, isTimeoutError } from '@/lib/utils/with-timeout'
import { safeLog } from '@/lib/utils/safe-log'
import { createServerClient } from '@/lib/supabase/server'

export const runtime = 'nodejs'
export const maxDuration = 30

export async function POST(request: NextRequest) {
  const requestId = request.headers.get('x-request-id') ?? crypto.randomUUID()

  try {
    const supabase = createServerClient(/* args */)

    // Auth: critical, must succeed
    const { data: { user } } = await withTimeout(
      supabase.auth.getUser(),
      5000,
      'api.payments.process.auth'
    )
    if (!user) {
      return NextResponse.json(
        { error: 'Authentication required', code: 'unauthenticated' },
        { status: 401 }
      )
    }

    // Body parsing
    const body = await request.json()

    // Database write: critical
    let result
    try {
      const queryResult = await withTimeout(
        supabase.from('payment_intents').insert(body).select().single(),
        15000,
        'api.payments.process.db'
      )
      if (queryResult.error) throw queryResult.error
      result = queryResult.data
    } catch (error) {
      if (isTimeoutError(error)) {
        safeLog.error('api.payments.process', 'database timeout', { userId: user.id, requestId, error })
        return NextResponse.json(
          { error: 'The payment system is temporarily slow. Please try again.', code: 'timeout' },
          { status: 503 }
        )
      }
      safeLog.error('api.payments.process', 'database error', { userId: user.id, requestId, error })
      return NextResponse.json(
        { error: 'Unable to process payment. Please try again.', code: 'server' },
        { status: 500 }
      )
    }

    safeLog.info('api.payments.process', 'payment intent created', {
      userId: user.id,
      paymentIntentId: result.id,
      requestId,
    })

    return NextResponse.json({ data: result }, { status: 200 })
  } catch (error) {
    safeLog.error('api.payments.process', 'unexpected error', { requestId, error })
    return NextResponse.json(
      { error: 'An unexpected error occurred.', code: 'server' },
      { status: 500 }
    )
  }
}

        recommendations = cached.data
      } catch {
        recommendations = null
      }
    }

    return NextResponse.json(
      {
        data: recommendations,
        stale: recommendations !== null && /* logic to check if from cache */ false,
      },
      { status: 200 }
    )
  } catch (error) {
    safeLog.error('api.recommendations.refresh', 'unexpected error', { requestId, error })
    return NextResponse.json({ error: 'Unable to refresh recommendations.', code: 'server' }, { status: 500 })

'use server'

import { withTimeout, isTimeoutError } from '@/lib/utils/with-timeout'
import { safeLog } from '@/lib/utils/safe-log'
import type { ActionResult } from '@/lib/actions/types'

interface CaqPhaseInput {
  phase: number
  data: Record<string, unknown>
}

export async function saveCaqPhase(input: CaqPhaseInput): Promise<ActionResult<{ phaseId: string; nextPhase: number }>> {
  try {
    const supabase = createServerClient(/* args */)

    const { data: { user } } = await withTimeout(
      supabase.auth.getUser(),
      5000,
      'caq.save-phase.auth'
    )
    if (!user) {
      return { success: false, error: 'Please sign in to save your progress.', code: 'unauthenticated' }
    }

    // Validation
    if (input.phase < 1 || input.phase > 7) {
      return { success: false, error: 'Invalid phase number.', code: 'validation' }
    }

    // Database write
    let phaseRecord
    try {
      const result = await withTimeout(
        supabase
          .from('caq_phase_responses')
          .upsert({ user_id: user.id, phase: input.phase, data: input.data })
          .select()
          .single(),
        15000,
        'caq.save-phase.db'
      )
      if (result.error) throw result.error
      phaseRecord = result.data
    } catch (error) {
      if (isTimeoutError(error)) {
        safeLog.error('caq.save-phase', 'database timeout', { userId: user.id, phase: input.phase, error })
        return {
          success: false,
          error: 'The save took too long. Your progress may not be saved. Please try again.',
          code: 'timeout',
        }
      }
      safeLog.error('caq.save-phase', 'database error', { userId: user.id, phase: input.phase, error })
      return { success: false, error: 'Unable to save your progress. Please try again.', code: 'server' }
    }

    safeLog.info('caq.save-phase', 'phase saved', {
      userId: user.id,
      phase: input.phase,
      phaseId: phaseRecord.id,
    })

    return {
      success: true,
      data: { phaseId: phaseRecord.id, nextPhase: input.phase + 1 },
    }
  } catch (error) {
    safeLog.error('caq.save-phase', 'unexpected error', { error })
    return { success: false, error: 'An unexpected error occurred. Please try again.', code: 'server' }
  }
}

Similar to Pattern E but on database/external failures, returns { success: true, data: <partial or null>, stale: true } instead of { success: false }. Only use this for actions where partial completion is genuinely useful to the caller.

Pattern for cron job Edge Functions (Category G):
Cron jobs run in the background and must not crash the entire job on a single failure. Use batch error handling with structured logging at start, milestones, and end.
typescript// supabase/functions/cron-bio-optimization-recalc/index.ts (illustrative)
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
import { withTimeout, isTimeoutError } from '../_shared/with-timeout.ts'
import { safeLog } from '../_shared/safe-log.ts'

serve(async (req) => {
  const jobId = crypto.randomUUID()
  const startedAt = new Date().toISOString()

  safeLog.info('cron.bio-optimization-recalc', 'job started', { jobId, startedAt })

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL') ?? '',
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') ?? ''
  )

  let processedUsers = 0
  let failedUsers = 0
  const failures: Array<{ userId: string; error: string }> = []

  try {
    // Fetch all eligible users
    const usersResult = await withTimeout(
      supabase.from('users').select('id').eq('caq_complete', true),
      30000,
      'cron.bio-optimization-recalc.fetch-users'
    )
    if (usersResult.error) throw usersResult.error
    const users = usersResult.data ?? []

    safeLog.info('cron.bio-optimization-recalc', 'users fetched', { jobId, userCount: users.length })

    // Process each user; one failure does not abort the job
    for (const user of users) {
      try {
        await withTimeout(
          recalculateBioOptimizationForUser(supabase, user.id),
          25000,
          'cron.bio-optimization-recalc.user-recalc'
        )
        processedUsers++
      } catch (error) {
        failedUsers++
        failures.push({
          userId: user.id,
          error: error instanceof Error ? error.message : String(error),
        })
        safeLog.warn('cron.bio-optimization-recalc', 'user recalc failed', {
          jobId,
          userId: user.id,
          error,
        })
        // Continue with next user
      }
    }

    safeLog.info('cron.bio-optimization-recalc', 'job complete', {
      jobId,
      startedAt,
      completedAt: new Date().toISOString(),
      processedUsers,
      failedUsers,
      failureSampleSize: Math.min(failures.length, 10),
      failureSample: failures.slice(0, 10),
    })

    return new Response(
      JSON.stringify({ success: true, processedUsers, failedUsers, jobId }),
      { status: 200, headers: { 'Content-Type': 'application/json' } }
    )
  } catch (error) {
    safeLog.error('cron.bio-optimization-recalc', 'job failed catastrophically', {
      jobId,
      startedAt,
      processedUsers,
      failedUsers,
      error,
    })
    return new Response(
      JSON.stringify({ success: false, error: 'Job failed', jobId }),
      { status: 500, headers: { 'Content-Type': 'application/json' } }
    )
  }

export class CircuitBreakerError extends Error {
  readonly breakerName: string
  constructor(breakerName: string) {
    super(`Circuit breaker "${breakerName}" is open`)
    this.name = 'CircuitBreakerError'
    this.breakerName = breakerName
  }
}

export interface CircuitBreakerOptions {
  failureThreshold: number
  resetTimeoutMs: number
  halfOpenMaxAttempts: number
}

const DEFAULT_OPTIONS: CircuitBreakerOptions = {
  failureThreshold: 5,
  resetTimeoutMs: 60000,
  halfOpenMaxAttempts: 1,
}

type State = 'closed' | 'open' | 'half-open'

export class CircuitBreaker {
  private state: State = 'closed'
  private failures = 0
  private successes = 0
  private lastFailureAt: Date | null = null
  private nextAttemptAt: Date | null = null
  private halfOpenAttempts = 0

  constructor(
    private readonly name: string,
    private readonly options: CircuitBreakerOptions = DEFAULT_OPTIONS
  ) {}

  async execute<T>(fn: () => Promise<T>): Promise<T> {
    this.maybeTransitionFromOpenToHalfOpen()

    if (this.state === 'open') {
      throw new CircuitBreakerError(this.name)
    }

    if (this.state === 'half-open' && this.halfOpenAttempts >= this.options.halfOpenMaxAttempts) {
      throw new CircuitBreakerError(this.name)
    }

    if (this.state === 'half-open') {
      this.halfOpenAttempts++
    }

    try {
      const result = await fn()
      this.onSuccess()
      return result
    } catch (error) {
      this.onFailure()
      throw error
    }
  }

  getState(): State {
    this.maybeTransitionFromOpenToHalfOpen()
    return this.state
  }

  getStats() {
    return {
      failures: this.failures,
      successes: this.successes,
      lastFailureAt: this.lastFailureAt,
    }
  }

  private maybeTransitionFromOpenToHalfOpen(): void {
    if (this.state === 'open' && this.nextAttemptAt && Date.now() >= this.nextAttemptAt.getTime()) {
      this.state = 'half-open'
      this.halfOpenAttempts = 0
    }
  }

  private onSuccess(): void {
    this.successes++
    if (this.state === 'half-open') {
      this.state = 'closed'
      this.failures = 0
      this.nextAttemptAt = null
    } else {
      this.failures = 0
    }
  }

  private onFailure(): void {
    this.failures++
    this.lastFailureAt = new Date()

    if (this.state === 'half-open') {
      this.state = 'open'
      this.nextAttemptAt = new Date(Date.now() + this.options.resetTimeoutMs)
    } else if (this.failures >= this.options.failureThreshold) {
      this.state = 'open'
      this.nextAttemptAt = new Date(Date.now() + this.options.resetTimeoutMs)
    }
  }
}

const breakers = new Map<string, CircuitBreaker>()

export function getCircuitBreaker(name: string, options?: CircuitBreakerOptions): CircuitBreaker {
  let breaker = breakers.get(name)
  if (!breaker) {
    breaker = new CircuitBreaker(name, options ?? DEFAULT_OPTIONS)
    breakers.set(name, breaker)
  }
  return breaker
}

export function isCircuitBreakerError(error: unknown): error is CircuitBreakerError {
  return error instanceof CircuitBreakerError
}
